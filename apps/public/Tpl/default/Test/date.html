請選擇日期： {:W('DateSelect',array('name'=>'dateSelect','class'=>'s-txt','id'=>'dateSelectId','value'=>'','dtype'=>'full'))} 
                    呼叫：W('DateSelect',array('name'=>'dateSelect','class'=>'s-txt','id'=>'dateSelectId','value'=>'','dtype'=>'full'))
                    參數說明：
                    name:元件input的名稱
                    class:元件input的樣式
                    id:元件input的ID
                    value:默認的值
                    dtype:時間類型
